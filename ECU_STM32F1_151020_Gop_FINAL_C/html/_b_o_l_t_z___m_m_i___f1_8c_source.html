178 void StartUpTask ( void *data )
186 {
187  BYTE err; // µC/OS error return code
188  BYTE iTask; // task number
189 
190  OS_CPU_SysTickInit(); // set desired uC/OS tick rate according
191  // to parameter OS_TICKS_PER_SEC in os_cfg.h
192 
193 #ifdef OS_STATISTICS_ON
194  OSStatInit(); // measure OSIdleCtr with NO USER TASK running
195  MaxIdleCount = OSIdleCtr;
196 #endif
197 
198  // Create user tasks
199  iTask = 1;
200  OSTaskCreate( UserTask_CANMsgSend, "UserTask_CANMsgSend",
201  (void *)&UserTaskStk[iTask-1][TASK_STK_SIZE-1], USER_BASEPRIO+iTask);

248 {

251 //UINT16 PotiScale100=0; // Pot value in the range of 0 to 100%
252  UINT16 ThrottleCmd1024=0; // temp new throttle value buffer
253  UINT16 ThrottleCmdOld = 0; // temp old throttle value buffer
254  UINT8 k = 5; // LPF smoothing factor
255 
256  while ( TRUE ) // DO FOREVER loop, uC/OS task NEVER returns
257  {
258  if ( on_off )
259  {
260  LED_on(STATUS_LED); // toggle STATUS_LED for each cycle (LED selected in platfrom_cfg.h)
261  on_off = FALSE;
262  }
263  else
264  {
265  LED_off(STATUS_LED);
266  on_off = TRUE;
267  }
268 
269  Potivalue = Get_Potvalue(); // raw poti position value [0 .. 4095]
270  ThrottleCmd1024 = Potivalue / 4; // scale to [0 .. 1023] range
271  ThrottleCmdNew = ((k-1) * ThrottleCmdOld + ThrottleCmd1024) / k; // low pass filter to smooth the throttlecmd value
272  ThrottleCmdOld = ThrottleCmdNew;
273 
274  #ifdef PRINT_ADValues
275  print_string( "AD=" ); // Printing Scaled ADC values

278  print_uint(PotiScale100);
